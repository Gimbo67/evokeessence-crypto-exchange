# Function to convert a folder to a group
convert_folder_to_group() {
  folder_name=$1
  # Find the folder reference
  folder_line=$(grep -n "path = $folder_name;" ./CryptoEvokeApp/CryptoEvokeApp.xcodeproj/project.pbxproj | grep -i "fileReference" | cut -d: -f1)

  if [ -n "$folder_line" ]; then
    # Get the reference ID
    reference_id=$(sed -n "${folder_line}p" ./CryptoEvokeApp/CryptoEvokeApp.xcodeproj/project.pbxproj | sed 's/^[ \t]*\([^ \t]*\).*/\1/')

    # Change isa = PBXFileReference to isa = PBXGroup
    sed -i '' "${folder_line}s/isa = PBXFileReference/isa = PBXGroup/" ./CryptoEvokeApp/CryptoEvokeApp.xcodeproj/project.pbxproj

    echo "Changed $folder_name to a group (reference ID: $reference_id)"
  else
    echo "Could not find folder reference for $folder_name"
  fi
}
