import React, { useState, useEffect } from 'react';
import { NavigationContainer } from '@react-navigation/native';
import { createStackNavigator } from '@react-navigation/stack';
import { 
  View, 
  Text, 
  TextInput,

} from 'react-native';

// API base URL - pointing to your live backend
const API_URL = 'https://evo-exchange.com/api';

// Create the stack navigator
const Stack = createStackNavigator();

// Login Screen Component
function LoginScreen({ navigation }) {
  const [username, setUsername] = useState('');
  const [password, setPassword] = useState('');
  const [isLoading, setIsLoading] = useState(false);
  const [error, setError] = useState(null);

  const handleLogin = async () => {
    console.log('Login button pressed');
    if (!username || !password) {
      Alert.alert('Error', 'Please enter both username and password');
      return;
    }

    setIsLoading(true);
    setError(null);

    try {
      const response = await axios.post(`${API_URL}/auth/login`, {
        username,
        password
      }, {
        headers: {
          'Content-Type': 'application/json',
          'X-App-Platform': 'ios'
        }
      });
      
      console.log('Login response:', JSON.stringify(response.data));
      
      // Check if 2FA is required
      if (response.data.requireTwoFactor) {
        // Save the username for the 2FA screen
        await AsyncStorage.setItem('temp_username', username);
        Alert.alert('Two-Factor Authentication', 'Please enter your verification code to continue.');
        navigation.navigate('TwoFactor', { username: username });
        setIsLoading(false);
        return;
      }
      
      // Store the entire response for debugging
      await AsyncStorage.setItem('last_login_response', JSON.stringify(response.data));
      
      // Store user data
      const userData = response.data.user || response.data;
      await AsyncStorage.setItem('user_data', JSON.stringify(userData));
      
      if (response.data.token) {
        await AsyncStorage.setItem('auth_token', response.data.token);
      }
      
      // Determine user type and navigate to the appropriate dashboard
      let userType = 'default';
      
      if (userData.isAdmin) {
        userType = 'admin';
      } else if (userData.isEmployee) {
        userType = 'employee';
      } else if (userData.verificationLevel >= 2) {
        userType = 'verified';
      } else {
        userType = 'unverified';
      }
      
      console.log(`Navigating to dashboard for user type: ${userType}`);
      await AsyncStorage.setItem('user_type', userType);
      
      // Force a hard navigation reset to ensure we leave the login screen
      setTimeout(() => {
        navigation.reset({
          index: 0,
          routes: [{ name: 'Dashboard', params: { userType } }],
        });
      }, 300);
    } catch (err) {
      console.error('Login error:', err.response?.data || err.message);
      setError(err.response?.data?.message || 'Login failed. Please check your credentials.');
      Alert.alert('Login Error', err.response?.data?.message || 'Login failed. Please check your credentials.');
    } finally {
      setIsLoading(false);
    }
  };

  return (
    <SafeAreaView style={styles.container}>
      <ScrollView contentContainerStyle={styles.scrollContent}>
        <View style={styles.header}>
          <View style={styles.logoContainer}>
            <Text style={styles.logoText}>EE</Text>
          </View>
          <Text style={styles.title}>EvokeEssence</Text>
          <Text style={styles.subtitle}>Login to your account</Text>
        </View>

        {error ? (
          <View style={styles.errorContainer}>
            <Text style={styles.errorText}>{error}</Text>
          </View>
        ) : null}

        <View style={styles.form}>
          <View style={styles.inputGroup}>
            <Text style={styles.label}>Username</Text>
            <TextInput
              style={styles.input}
              placeholder="Enter your username"
              value={username}
              onChangeText={setUsername}
              autoCapitalize="none"
              autoCorrect={false}
            />
          </View>

          <View style={styles.inputGroup}>
            <Text style={styles.label}>Password</Text>
            <TextInput
              style={styles.input}
              placeholder="Enter your password"
              value={password}
              onChangeText={setPassword}
              secureTextEntry
            />
          </View>

          <TouchableOpacity
            style={styles.loginButton}
            onPress={handleLogin}
            disabled={isLoading}
          >
            {isLoading ? (
              <ActivityIndicator color="#fff" size="small" />
            ) : (
              <Text style={styles.loginButtonText}>Login</Text>
            )}
          </TouchableOpacity>

          <View style={styles.registerContainer}>
            <Text style={styles.registerText}>Don't have an account? </Text>
            <TouchableOpacity onPress={() => navigation.navigate('Register')}>
              <Text style={styles.registerLink}>Register</Text>
            </TouchableOpacity>
          </View>
        </View>
      </ScrollView>
    </SafeAreaView>
  );
}

// Register Screen Component
function RegisterScreen({ navigation }) {
  const [formData, setFormData] = useState({
    username: '',
    email: '',
    password: '',
    confirmPassword: '',
    firstName: '',
    lastName: '',
    phone: '',
    country: '',
    referralCode: ''
  });
  const [isLoading, setIsLoading] = useState(false);
  const [error, setError] = useState(null);

  const handleChange = (field, value) => {
    setFormData({ ...formData, [field]: value });
  };

  const handleRegister = async () => {
    // Basic validation
    if (!formData.username || !formData.email || !formData.password) {
      Alert.alert('Error', 'Please fill in all required fields');
      return;
    }

    if (formData.password !== formData.confirmPassword) {
      Alert.alert('Error', 'Passwords do not match');
      return;
    }

    setIsLoading(true);
    setError(null);

    try {
      const response = await axios.post(`${API_URL}/auth/register`, {
        username: formData.username,
        email: formData.email,
        password: formData.password,
        first_name: formData.firstName,
        last_name: formData.lastName,
        phone: formData.phone,
        country: formData.country,
        referral_code: formData.referralCode
      }, {
        headers: {
          'Content-Type': 'application/json',
          'X-App-Platform': 'ios'
        }
      });

        'Registration Successful', 
        'Your account has been created. Please check your email for verification instructions.',

// Dashboard Screen Component
function DashboardScreen({ navigation, route }) {
  const [user, setUser] = useState(null);
  const [userType, setUserType] = useState('loading');
  const [isLoading, setIsLoading] = useState(true);
  const [debugInfo, setDebugInfo] = useState('Initializing dashboard...');

  useEffect(() => {
    console.log('Dashboard screen mounted with params:', route.params);
    
    // Load user data from storage
    const loadUserData = async () => {
      try {
        setDebugInfo('Checking AsyncStorage for user data...');
        const userData = await AsyncStorage.getItem('user_data');
        const storedUserType = await AsyncStorage.getItem('user_type') || route.params?.userType || 'default';
        const lastLoginResponse = await AsyncStorage.getItem('last_login_response');
        
        console.log('Last login response:', lastLoginResponse);
        console.log('User type from storage/params:', storedUserType);
        
        setUserType(storedUserType);
        
        if (userData) {
          try {
            const parsedData = JSON.parse(userData);
            console.log('User data loaded from storage:', JSON.stringify(parsedData));
            setDebugInfo(`User data loaded: ${JSON.stringify(parsedData).substring(0, 50)}...`);
            setUser(parsedData);
            setIsLoading(false);
          } catch (parseError) {
            console.error('Error parsing user data:', parseError);
            setDebugInfo(`Error parsing user data: ${parseError.message}`);
            // Try to get session from server instead
            fetchSessionData();
          }
        } else {
          setDebugInfo('No user data in AsyncStorage, trying session check...');
          // If no user data found, try to get session
          fetchSessionData();
        }
      } catch (error) {
        console.error('Error loading user data:', error);
        setDebugInfo(`Error loading user data: ${error.message}`);
        fetchSessionData();
      }
    };

    const fetchSessionData = async () => {
      try {
        setDebugInfo('Fetching session from server...');
        const response = await axios.get(`${API_URL}/auth/session`, {
          headers: {
            'Content-Type': 'application/json',
            'X-App-Platform': 'ios'
          }
        });
        
        console.log('Session response:', JSON.stringify(response.data));
        
        if (response.data && response.data.authenticated) {
          setDebugInfo('Session authenticated, storing user data');
          
          const userData = response.data.user;
          setUser(userData);
          await AsyncStorage.setItem('user_data', JSON.stringify(userData));
          
          // Determine user type
          let detectedUserType = 'default';
          if (userData.isAdmin) {
            detectedUserType = 'admin';
          } else if (userData.isEmployee) {
            detectedUserType = 'employee';
          } else if (userData.verificationLevel >= 2) {
            detectedUserType = 'verified';
          } else {
            detectedUserType = 'unverified';
          }
          
          setUserType(detectedUserType);
          await AsyncStorage.setItem('user_type', detectedUserType);
          
          setIsLoading(false);
        } else {
          setDebugInfo('Session not authenticated, redirecting to login');
          // Not authenticated, go back to login
          redirectToLogin();
        }
      } catch (err) {
        console.error('Session check error:', err);
        setDebugInfo(`Session check error: ${err.message}`);
        redirectToLogin();
      }
    };

    const redirectToLogin = () => {
      // Delay to allow seeing debug info
      setTimeout(() => {
        navigation.reset({
          index: 0,
          routes: [{ name: 'Login' }],
        });
      }, 1000);
    };

    loadUserData();
  }, [navigation, route.params]);

  const handleLogout = async () => {
    setIsLoading(true);
    setDebugInfo('Logging out...');
    
    try {
      await axios.post(`${API_URL}/auth/logout`, {}, {
        headers: {
          'Content-Type': 'application/json',
          'X-App-Platform': 'ios'
        }
      });
      console.log('Logout successful');
      setDebugInfo('Logout successful, clearing storage');
    } catch (err) {
      console.error('Logout error:', err);
      setDebugInfo(`Logout error: ${err.message}`);
    } finally {
      // Clear all storage
      await AsyncStorage.removeItem('user_data');
      await AsyncStorage.removeItem('auth_token');
      await AsyncStorage.removeItem('temp_username');
      await AsyncStorage.removeItem('user_type');
      await AsyncStorage.removeItem('last_login_response');
      
      setTimeout(() => {
        navigation.reset({
          index: 0,
          routes: [{ name: 'Login' }],
        });
      }, 300);
    }
  };

  if (isLoading) {
    return (
      <View style={styles.loadingContainer}>
        <ActivityIndicator size="large" color="#6200ee" />
        <Text style={styles.loadingText}>Loading dashboard...</Text>
        <Text style={styles.debugText}>{debugInfo}</Text>
      </View>
    );
  }

  const renderDashboardContent = () => {
    switch (userType) {
      case 'admin':
        return (
          <View style={styles.dashboardContent}>
            <Text style={styles.userTypeLabel}>Admin Dashboard</Text>
            <View style={styles.adminCard}>
              <Text style={styles.cardTitle}>Admin Controls</Text>
              <TouchableOpacity style={styles.adminButton}>
                <Text style={styles.adminButtonText}>Manage Users</Text>
              </TouchableOpacity>
              <TouchableOpacity style={styles.adminButton}>
                <Text style={styles.adminButtonText}>View Analytics</Text>
              </TouchableOpacity>
              <TouchableOpacity style={styles.adminButton}>
                <Text style={styles.adminButtonText}>Security Settings</Text>
              </TouchableOpacity>
            </View>
          </View>
        );
        
      case 'employee':
        return (
          <View style={styles.dashboardContent}>
            <Text style={styles.userTypeLabel}>Employee Dashboard</Text>
            <View style={styles.employeeCard}>
              <Text style={styles.cardTitle}>Client Management</Text>
              <TouchableOpacity style={styles.employeeButton}>
                <Text style={styles.employeeButtonText}>View Clients</Text>
              </TouchableOpacity>
              <TouchableOpacity style={styles.employeeButton}>
                <Text style={styles.employeeButtonText}>Pending Verifications</Text>
              </TouchableOpacity>
            </View>
          </View>
        );
        
      case 'verified':
        return (
          <View style={styles.dashboardContent}>
            <Text style={styles.userTypeLabel}>Verified Client Dashboard</Text>
            <View style={styles.accountCard}>
              <Text style={styles.cardTitle}>Account Overview</Text>
              <View style={styles.balanceRow}>
                <Text style={styles.balanceLabel}>Available Balance</Text>
                <Text style={styles.balanceValue}>$0.00</Text>
              </View>
              
              <TouchableOpacity style={styles.buyButton}>
                <Text style={styles.buyButtonText}>Buy Crypto</Text>
              </TouchableOpacity>
            </View>
            
            <View style={styles.actionsCard}>
              <Text style={styles.cardTitle}>Quick Actions</Text>
              <View style={styles.actionButtonsRow}>
                <TouchableOpacity style={styles.actionButton}>
                  <Text style={styles.actionButtonText}>Deposit</Text>
                </TouchableOpacity>
                <TouchableOpacity style={styles.actionButton}>
                  <Text style={styles.actionButtonText}>Withdraw</Text>
                </TouchableOpacity>
                <TouchableOpacity style={styles.actionButton}>
                  <Text style={styles.actionButtonText}>Trade</Text>
                </TouchableOpacity>
              </View>
            </View>
          </View>
        );
        
      case 'unverified':
        return (
          <View style={styles.dashboardContent}>
            <Text style={styles.userTypeLabel}>Complete Your Verification</Text>
            <View style={styles.verificationCard}>
              <Text style={styles.cardTitle}>Verification Required</Text>

                To access all features, please complete your identity verification.
              </Text>

// Two Factor Screen Component
function TwoFactorScreen({ navigation, route }) {
  const [code, setCode] = useState('');
  const [isLoading, setIsLoading] = useState(false);
  const [error, setError] = useState(null);
  const [debugInfo, setDebugInfo] = useState('Enter verification code');
  
  useEffect(() => {
    const checkUsername = async () => {
      const storedUsername = await AsyncStorage.getItem('temp_username');
      setDebugInfo(`Username from storage: ${storedUsername || 'none'}, from params: ${route.params?.username || 'none'}`);
    };
    
    checkUsername();
  }, [route.params]);
  
  const handleVerify = async () => {
    if (!code.trim()) {
      Alert.alert('Error', 'Please enter the verification code');
      return;
    }

    setIsLoading(true);
    setError(null);

    try {
      // Get the username from route params or from AsyncStorage
      const username = route.params?.username || await AsyncStorage.getItem('temp_username');
      
      if (!username) {
        setError('Username not found. Please go back to login.');
        setIsLoading(false);
        return;
      }
      
      console.log('Verifying 2FA for username:', username, 'with code:', code);
      setDebugInfo(`Verifying 2FA for ${username} with code ${code}`);
      
      const response = await axios.post(`${API_URL}/auth/verify-2fa`, {
        code,
        username
      }, {
        headers: {
          'Content-Type': 'application/json',
          'X-App-Platform': 'ios'
        }
      });
      
      console.log('2FA verification response:', response.data);
      setDebugInfo(`2FA verification successful: ${JSON.stringify(response.data).substring(0, 50)}...`);
      
      // Store the entire response for debugging
      await AsyncStorage.setItem('last_login_response', JSON.stringify(response.data));
      
      // Store user data
      const userData = response.data.user || response.data;
      await AsyncStorage.setItem('user_data', JSON.stringify(userData));
      
      if (response.data.token) {
        await AsyncStorage.setItem('auth_token', response.data.token);
      }
      
      // Determine user type
      let userType = 'default';
      if (userData.isAdmin) {
        userType = 'admin';
      } else if (userData.isEmployee) {
        userType = 'employee';
      } else if (userData.verificationLevel >= 2) {
        userType = 'verified';
      } else {
        userType = 'unverified';
      }
      
      await AsyncStorage.setItem('user_type', userType);
      
      // Remove temp username
      await AsyncStorage.removeItem('temp_username');
      
      // Navigate to dashboard with reset
      setTimeout(() => {
        navigation.reset({
          index: 0,
          routes: [{ name: 'Dashboard', params: { userType } }],
        });
      }, 300);
    } catch (err) {
      console.error('2FA verification error:', err.response?.data || err.message);
      setDebugInfo(`2FA verification error: ${err.message}`);
      setError(err.response?.data?.message || 'Invalid verification code');
      Alert.alert('Verification Error', err.response?.data?.message || 'Invalid verification code');
    } finally {
      setIsLoading(false);
    }
  };

  return (
    <SafeAreaView style={styles.container}>
      {/* Debug Info */}
      <View style={styles.debugContainer}>
        <Text style={styles.debugText}>{debugInfo}</Text>
      </View>
      
      <View style={styles.content}>
        <View style={styles.header}>
          <View style={styles.logoContainer}>
            <Text style={styles.logoText}>EE</Text>
          </View>
          <Text style={styles.title}>Two-Factor Authentication</Text>
          <Text style={styles.subtitle}>
            Please enter the 6-digit code from your authenticator app
          </Text>
        </View>

        {error ? (
          <View style={styles.errorContainer}>
            <Text style={styles.errorText}>{error}</Text>
          </View>
        ) : null}

        <View style={styles.form}>
          <View style={styles.inputGroup}>
            <Text style={styles.label}>Verification Code</Text>
            <TextInput
              style={[styles.input, styles.codeInput]}
              placeholder="Enter 6-digit code"
              value={code}
              onChangeText={setCode}
              keyboardType="number-pad"
              maxLength={6}
              autoFocus={true}
            />
          </View>

          <TouchableOpacity
            style={styles.loginButton}
            onPress={handleVerify}
            disabled={isLoading}
          >
            {isLoading ? (
              <ActivityIndicator color="#fff" size="small" />
            ) : (
              <Text style={styles.loginButtonText}>Verify</Text>
            )}
          </TouchableOpacity>

          <View style={styles.registerContainer}>
            <TouchableOpacity onPress={() => navigation.navigate('Login')}>
              <Text style={styles.registerLink}>Back to Login</Text>
            </TouchableOpacity>
          </View>
        </View>
      </View>
    </SafeAreaView>
  );
}
